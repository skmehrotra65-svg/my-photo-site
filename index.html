<script>
/* ============================================================
CONTENT NOW LIVES IN data/photos.json AND data/site.json
------------------------------------------------------------
Edit those files directly, or — once the CMS is set up at
/admin — edit them through the browser instead. This page
just fetches them and renders the result.
Note: fetch() of local files is blocked by browsers when you
open index.html directly by double-clicking it. To see this
working, view the site through Netlify (or run a local server).
============================================================ */

let photos = [];
const DEFAULT_GRADIENT = 'linear-gradient(155deg,#2a2a20,#1b1b14)';

const grid = document.getElementById('grid');
const filters = document.getElementById('filters');
const regionFilters = document.getElementById('regionFilters');
let activeFilter = 'all';
let activeRegion = 'all';

// Routes images through Netlify's Image CDN: serves WebP/AVIF automatically based on
// the visitor's browser, resizes on the fly, and falls back to the original file if
// the site isn't running on Netlify yet (e.g. while testing locally).
function optimizedSrc(path, width){
return `/.netlify/images?url=${encodeURIComponent('/' + path)}&w=${width}&q=80`;
}

function cardMedia(p){
if(p.src){
return `<img src="${optimizedSrc(p.src, 800)}" onerror="this.onerror=null;this.src='${p.src}';" alt="${p.title} — ${p.location}" loading="lazy" draggable="false">`;
}
return `<span class="placeholder-tag">No image — add to /images</span>`;
}

function renderGrid(){
const visible = photos.filter(p =>
(activeFilter === 'all' || p.category === activeFilter) &&
(activeRegion === 'all' || p.region === activeRegion)
);
grid.innerHTML = visible.map((p) => {
const realIndex = photos.indexOf(p);
return `
<article class="card" tabindex="0" data-index="${realIndex}">
<div class="card-media" style="aspect-ratio:${p.width || 1400}/${p.height || 1000}; ${p.src ? '' : `background:${p.gradient || DEFAULT_GRADIENT}`}">
${cardMedia(p)}
</div>
<div class="card-note">
<div class="card-note-top">
<span class="card-category cat-${p.category}">${p.category}</span>
<span class="card-region">${p.region}</span>
</div>
<div class="card-meta">
<span>${p.location}</span>
<span>${p.coords}</span>
<span>${p.date} · ${p.settings}</span>
</div>
</div>
</article>`;
}).join('');

grid.querySelectorAll('.card').forEach(card => {
card.addEventListener('click', () => openLightbox(parseInt(card.dataset.index)));
card.addEventListener('keypress', (e) => { if(e.key === 'Enter') openLightbox(parseInt(card.dataset.index)); });
});

layoutMasonry();
}

/* ---------- true masonry: measure each card's real rendered height and
tell the grid how many fine-grained rows it should span ---------- */
function layoutMasonry(){
const styles = getComputedStyle(grid);
const rowHeight = parseFloat(styles.gridAutoRows) || 1;
const rowGap = parseFloat(styles.rowGap || styles.gridRowGap) || 0;
grid.querySelectorAll('.card').forEach(card => {
const media = card.querySelector('.card-media');
const h = (media || card).getBoundingClientRect().height;
const span = Math.ceil((h + rowGap) / (rowHeight + rowGap));
card.style.gridRowEnd = `span ${span}`;
});
}

let resizeTimer;
window.addEventListener('resize', () => {
clearTimeout(resizeTimer);
resizeTimer = setTimeout(layoutMasonry, 150);
});
if(document.fonts && document.fonts.ready){ document.fonts.ready.then(layoutMasonry); }

/* ---------- category + region filter pills, built automatically from whatever
values exist in the photo data — add a new category or region in the CMS
and a matching filter button just appears, no code involved ---------- */
function buildPillFilters(container, key, activeKeyName, onPick, allLabel){
const values = ['all', ...new Set(photos.map(p => p[key]).filter(Boolean))];
container.innerHTML = values.map(v =>
`<button data-value="${v}" class="${v === 'all' ? 'active' : ''}">${v === 'all' ? allLabel : v}</button>`
).join('');
container.querySelectorAll('button').forEach(btn => {
btn.addEventListener('click', () => {
container.querySelectorAll('button').forEach(b => b.classList.remove('active'));
btn.classList.add('active');
onPick(btn.dataset.value);
renderGrid();
});
});
}

/* ---------- apply editable site text from data/site.json ---------- */
function applySiteData(site){
if(!site) return;
if(site.heroTitle) document.getElementById('heroTitle').textContent = site.heroTitle;
if(site.name || site.tagline){
document.getElementById('heroSub').textContent = site.tagline || '';
}
document.querySelectorAll('.site-name').forEach(el => { if(site.name) el.textContent = site.name; });
if(site.aboutHeading) document.getElementById('aboutHeading').textContent = site.aboutHeading;
if(site.aboutParagraph1) document.getElementById('aboutP1').textContent = site.aboutParagraph1;
if(site.aboutParagraph2) document.getElementById('aboutP2').textContent = site.aboutParagraph2;
if(site.pullQuote) document.getElementById('pullQuote').textContent = `"${site.pullQuote}"`;
if(Array.isArray(site.stats) && site.stats.length){
document.getElementById('statRow').innerHTML = site.stats.map(s =>
`<div><span class="num">${s.num}</span><span class="label">${s.label}</span></div>`
).join('');
}
if(site.contactEmail){
const emailEl = document.getElementById('contactEmail');
emailEl.textContent = site.contactEmail;
emailEl.href = `mailto:${site.contactEmail}`;
}
if(Array.isArray(site.socialLinks) && site.socialLinks.length){
document.getElementById('socialRow').innerHTML = site.socialLinks.map(s =>
`<a href="${s.url}" target="_blank" rel="noopener">${s.label}</a>`
).join('');
}
if(site.heroImage){
const heroEl = document.querySelector('.hero');
heroEl.style.backgroundImage =
`linear-gradient(180deg, rgba(15,15,11,.45) 0%, rgba(15,15,11,.65) 55%, rgba(15,15,11,.92) 100%), url('${optimizedSrc(site.heroImage, 1800)}')`;
heroEl.style.backgroundSize = 'cover';
heroEl.style.backgroundPosition = 'center';
}
}

/* ---------- load everything, then render ---------- */
async function init(){
try{
const [photosRes, siteRes] = await Promise.all([
fetch('data/photos.json'),
fetch('data/site.json')
]);
const photosData = await photosRes.json();
const siteData = await siteRes.json();
photos = photosData.photos || [];
applySiteData(siteData);
}catch(err){
console.error('Could not load site content:', err);
grid.innerHTML = '<p style="font-family:var(--font-mono); font-size:.85rem; color:var(--text-dim);">Couldn\'t load photo data. Make sure data/photos.json exists and you\'re viewing this through Netlify, not as a local file.</p>';
}

buildPillFilters(filters, 'category', 'activeFilter', (v) => activeFilter = v, 'All');
buildPillFilters(regionFilters, 'region', 'activeRegion', (v) => activeRegion = v, 'All locations');
renderGrid();

const tickerStrip = document.getElementById('tickerStrip');
const tickerItems = photos.map(p => `${p.location} <span class="sep">·</span> ${p.coords}`);
tickerStrip.innerHTML = tickerItems.concat(tickerItems).map(t => `<span class="t-item">${t}</span>`).join('<span class="sep"> — </span>');

/* ---------- hero slideshow — loaded from data/slideshow.json ---------- */
async function initSlideshow(){
try{
const res = await fetch('data/slideshow.json');
const data = await res.json();
const slides = (data.slides || []).filter(s => s.src);
const slidesEl = document.getElementById('heroSlides');
if(!slides.length) return;
slides.forEach((s, i) => {
const div = document.createElement('div');
div.className = 'hero-slide' + (i === 0 ? ' active' : '');
div.style.backgroundImage = `url('${optimizedSrc(s.src, 1800)}')`;
div.setAttribute('aria-hidden','true');
slidesEl.appendChild(div);
});
let current = 0;
setInterval(() => {
const els = slidesEl.querySelectorAll('.hero-slide');
els[current].classList.remove('active');
current = (current + 1) % els.length;
els[current].classList.add('active');
}, 5000);
}catch(e){ console.warn('Could not load slideshow.json', e); }
}
initSlideshow();
}
init();

/* ---------- lightbox ---------- */
const lightbox = document.getElementById('lightbox');
const lbMedia = document.getElementById('lbMedia');
const lbCaption = document.getElementById('lbCaption');
let currentIndex = 0;

function openLightbox(index){
currentIndex = index;
lastFocusedCard = document.activeElement;
renderLightbox();
lightbox.classList.add('open');
}
function renderLightbox(){
const p = photos[currentIndex];
lbMedia.style.background = p.src ? 'transparent' : p.gradient;
lbMedia.innerHTML = p.src
? `<img src="${optimizedSrc(p.src, 1800)}" onerror="this.onerror=null;this.src='${p.src}';" alt="${p.title} — ${p.location}" draggable="false">`
: `<span class="placeholder-tag" style="position:relative; display:inline-block; margin:20px;">No image yet — add ${p.title.toLowerCase().replace(/[^a-z0-9]+/g,'-')}.jpg to /images</span>`;
lbCaption.innerHTML = `
<h3>${p.title}</h3>
<span class="card-category cat-${p.category}">${p.category}</span>
<span>${p.location} · ${p.coords}</span>
<span>${p.date} · ${p.settings}</span>`;
}
function closeLightbox(){
lightbox.classList.remove('open');
if(lastFocusedCard) lastFocusedCard.focus();
}
function nextPhoto(){ currentIndex = (currentIndex + 1) % photos.length; renderLightbox(); }
function prevPhoto(){ currentIndex = (currentIndex - 1 + photos.length) % photos.length; renderLightbox(); }

let lastFocusedCard = null;
let touchStartX = 0;
lightbox.addEventListener('touchstart', (e) => { touchStartX = e.changedTouches[0].clientX; });
lightbox.addEventListener('touchend', (e) => {
const dx = e.changedTouches[0].clientX - touchStartX;
if(Math.abs(dx) > 50){ dx < 0 ? nextPhoto() : prevPhoto(); }
});

document.getElementById('lbClose').addEventListener('click', closeLightbox);
document.getElementById('lbNext').addEventListener('click', nextPhoto);
document.getElementById('lbPrev').addEventListener('click', prevPhoto);
lightbox.addEventListener('click', (e) => { if(e.target === lightbox) closeLightbox(); });
document.addEventListener('keydown', (e) => {
if(!lightbox.classList.contains('open')) return;
if(e.key === 'Escape') closeLightbox();
if(e.key === 'ArrowRight') nextPhoto();
if(e.key === 'ArrowLeft') prevPhoto();
});

/* ---------- nav scroll state ---------- */
const siteNav = document.getElementById('siteNav');
window.addEventListener('scroll', () => {
siteNav.classList.toggle('scrolled', window.scrollY > 40);
});

/* ---------- mobile menu ---------- */
const menuBtn = document.getElementById('menuBtn');
const mobileMenu = document.getElementById('mobileMenu');
const mobileMenuClose = document.getElementById('mobileMenuClose');
menuBtn.addEventListener('click', () => mobileMenu.classList.add('open'));
mobileMenuClose.addEventListener('click', () => mobileMenu.classList.remove('open'));
mobileMenu.querySelectorAll('a').forEach(a => a.addEventListener('click', () => mobileMenu.classList.remove('open')));

/* ---------- scroll reveal ---------- */
const revealObserver = new IntersectionObserver((entries) => {
entries.forEach(entry => {
if(entry.isIntersecting){
entry.target.classList.add('in-view');
revealObserver.unobserve(entry.target);
}
});
}, { threshold: 0.15 });
document.querySelectorAll('.reveal').forEach(el => revealObserver.observe(el));

/* ---------- footer year ---------- */
document.getElementById('year').textContent = new Date().getFullYear();

/* ---------- basic deterrents against right-click-save / dragging photos ----------
Note: this stops casual saving, it does not make images un-downloadable —
see the photographer notes for the protections that actually matter. */
document.addEventListener('contextmenu', (e) => { if(e.target.tagName === 'IMG') e.preventDefault(); });
document.addEventListener('dragstart', (e) => { if(e.target.tagName === 'IMG') e.preventDefault(); });
